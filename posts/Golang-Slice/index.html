func main() {
	arr := [...]int{1,2,3,4,5}

	slice1 := arr[:]
	fmt.Printf("%p %p %d %d \n", &slice1, slice1, len(slice1), cap(slice1))

	fmt.Printf("%p %p %d %d \n", &slice2, slice2, len(slice2), cap(slice2))
